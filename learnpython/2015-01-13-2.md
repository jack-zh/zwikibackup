    i, j, n = 0, 0, 5
    for i in range(1,n):
	  for j in range(n-i):
		  print(' ',end = '')
		  j += 1
	  for j in range(2*i - 1):
		  print('*',end = '')
		  j += 1
	  print()
	  i +=1

	  if i == n:
		  while i>=0:
			  for j in range(n-i):
				  print(' ',end = '')
				  j -= 1
			  for j in range(2*i - 1):
				  print('*',end = '')
				  j -= 1
			  print()
			  i -= 1
